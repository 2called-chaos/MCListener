  488   public function observe($file)
  489   {
  490     if(!file_exists($file)) {
  491       $this->log('The file "' . $file . '" was not found');
  492       throw new Exception('The file "' . $file . '" was not found');
  493     }
  494 
  495     $this->file = $file;
  496     $this->handle = fopen($this->file, "r");
  497     $this->mtime = filemtime($this->file);
  498     $this->size = filesize($this->file);
  499 
  500     $this->log('MCListener ' . self::VERSION . ' started');
  501     $this->_watchLog();
  502 
  503     return $this;
  504   }

  506   protected function _watchLog()
  507   {
  508     while (true) {
  509       clearstatcache();
  510       $this->cmtime = filemtime($this->file);
  511       $this->csize = filesize($this->file);
  512 
  513       // nothing changed, sleep and wait for new data
  514       if ($this->mtime == $this->cmtime) {
  515         usleep($this->delay);
  516         continue;
  517       }
  518 
  519       // changed, open file and get new data
  520       $this->newdata = '';
  521 
  522       // seek to position and gt new data
  523       fseek($this->handle, $this->size);
  524       while ($data = fgets($this->handle)) {
  525         $this->newdata .= $data;
  526       }
  527 
  528       // update values
  529       $this->size = $this->csize;
  530       $this->mtime = $this->cmtime;
  531 
  532       // process new data
  533       $this->_processData();
  534     }
  535   }
  536 
  537   protected function _processData()
  538   {
  539     // preparation
  540     $data = trim($this->newdata);
  541     $chunks = explode("\n", $data);
  542 
  543     foreach($chunks as $chunk) {
  544       $re1='(?:(?:2|1)\\d{3}(?:-|\\/)(?:(?:0[1-9])|(?:1[0-2]))(?:-|\\/)(?:(?:0[1-9])|(?:[1-2][0-9])|(?:3[0-1]))(?:T|\\s)(?:(?:[0-1][0-9])|(?:2[0-3])):(?:[0-5][0-9]):(?:[0-5][0-9]))';  # Timestamp
  545       $re2='(?:\\s+\\[INFO\\]\\s+)';  # square braces
  546       $re3='(<[^>]+>)'; # tag resp. player
  547       $re4='(?:\\s+)';  # white space
  548       $re5='((?:.+))';  # command
  549 
  550       if(preg_match_all("/".$re1.$re2.$re3.$re4.$re5."/is", $chunk, $matches)) {
  551         $user = str_replace(array('<', '>'), '', $matches[1][0]);
  552         $cmd = $matches[2][0];
  553 
  554         if(substr($cmd, 0, strlen($this->prefix)) == $this->prefix) {
  555           $raw = trim(substr($cmd, strlen($this->prefix)));
  556           $split = explode(" ", $raw);
  557           $cmd = array_shift($split);
  558           $this->cmd($user, $cmd, $split);
  559         }
  560       }
  561     }
  562   }

  571   public function give($user, $item, $amount = null)
  572   {
  573     // block bedrock
  574     if($item == 7) {
  575       $this->pm($user, 'no, not this one ;)');
  576       return;
  577     }
  578     
  579     // detect kit
  580     if(substr($item, 0, 1) == '#') {
  581       $this->giveKit($user, $item);
  582       return;
  583     }
  584     
  585     // default amount if not set
  586     if(is_null($amount)) {
  587       $amount = $this->defaultGiveAmount;
  588     }
  589     
  590     // maximum amount
  591     $limitExceeded = false;
  592     if($amount > 2304) {
  593       $amount = 2304;
  594       $limitExceeded = true;
  595     }
  596 
  597     // give items
  598     $times = floor($amount / 64);
  599     $lamount = $amount - ($times * 64);
  600 
  601     for ($i=0; $i < $times; $i++) {
  602       $this->mcexec('give ' . $user . ' ' . $item . ' 64');
  603     }
  604 
  605     if($lamount) {
  606       $this->mcexec('give ' . $user . ' ' . $item . ' ' . $lamount);
  607     }
  608 
  609     // send limit exceeded message if necessary 
  610     if($limitExceeded) {
  611       $this->pm($user, 'The amount is to high, you will get the maximum of 2304 items!');
  612     }
  613 
  614     return $this;
  615   }

  617   public function giveKit($user, $kit)
  618   {
  619     $items = array();
  620     
  621     // parse kit variable
  622     $kit = substr($kit, 1);
  623     $chunks = explode('+', $kit);
  624     
  625     foreach($chunks as $c) {
  626       $c = trim($c);
  627       $val = explode(':', $c);
  628       $items[$val[0]] = $val[1];
  629     }
  630     
  631     // give items
  632     foreach($items as $item => $amount) {
  633       if(is_numeric($item)) {
  634         $this->give($user, $item, $amount);
  635       } elseif (array_key_exists($item, $this->itemmap)) {
  636         $this->give($user, $this->itemmap[$item], $amount);
  637       } else {
  638         $this->pm($user, 'This item ID in this kit is wrong.');
  639       }
  640     }
  641   }

  660   public function cmd($user, $cmd, $params = array())
